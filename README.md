```bash
# @antfu/ni
ni @kay/ts-utils

# bun
bun add @kay/ts-utils
# pnpm
pnpm add @kay/ts-utils

# yarn
yarn install @kay/ts-utils
# npm
npm install @kay/ts-utils
```
